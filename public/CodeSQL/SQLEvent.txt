CREATE EVENT IF NOT EXISTS Update_Status2_3
ON SCHEDULE EVERY 1 MINUTE	
DO
UPDATE sortie SET etat_id =3 WHERE CURRENT_TIMESTAMP > date_limite_inscription AND etat_id=2;
CREATE EVENT IF NOT EXISTS Update_Status3_4
ON SCHEDULE EVERY 1 MINUTE	
DO
UPDATE sortie SET etat_id =4 WHERE CURRENT_TIMESTAMP > date_heure_debut AND etat_id=3;
CREATE EVENT IF NOT EXISTS Update_Status4_5
ON SCHEDULE EVERY 1 MINUTE	
DO
UPDATE sortie SET etat_id =5 WHERE CURRENT_TIMESTAMP > (DATE_ADD(date_heure_debut, INTERVAL duree MINUTE)) AND etat_id=4;
